Sub 右隣セルの文字列を結合()
    '選択したセルの文字列に右隣セルの文字列を結合（改行して結合）
    Dim Y As Integer
    Dim X As Integer
    Dim XR As Integer
    Dim i As Integer
    Dim myRange As Range
    
    XR = Application.InputBox(Prompt:="選択中のセルに右側何セル分の文字列を結合しますか？", Type:=1)

    For Each myRange In Selection.SpecialCells(xlCellTypeVisible)
        Y = myRange.Row
        X = myRange.Column
        For i = 1 To XR
            myRange.Value = myRange.Value & vbCrLf & Cells(Y, X + i).Value
        Next i
    Next myRange
    
End Sub
