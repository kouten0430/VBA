Sub 行全体を網かけ()
    Dim myRange As Range
    
    For Each myRange In Selection
        myRange.EntireRow.Interior.Pattern = xlPatternGray8
        
    Next myRange

End Sub
Sub 行全体の網かけをクリア()
    Dim myRange As Range
    
    For Each myRange In Selection
        myRange.EntireRow.Interior.Pattern = xlPatternNone
        
    Next myRange

End Sub
Sub 列全体を網かけ()
    Dim myRange As Range
    
    For Each myRange In Selection
        myRange.EntireColumn.Interior.Pattern = xlPatternGray8
        
    Next myRange

End Sub
Sub 列全体の網かけをクリア()
    Dim myRange As Range
    
    For Each myRange In Selection
        myRange.EntireColumn.Interior.Pattern = xlPatternNone
        
    Next myRange

End Sub
