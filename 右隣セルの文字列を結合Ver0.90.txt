Sub 右隣セルの文字列を結合()
    '選択したセルの文字列に右隣セルの文字列を結合（改行して結合）
    Dim Y As Long
    Dim X As Long
    Dim myRange As Range

    For Each myRange In Selection.SpecialCells(xlCellTypeVisible)
        Y = myRange.Row
        X = myRange.Column
        myRange.Value = myRange.Value & vbCrLf & Cells(Y, X + 1).Value
    Next myRange
    
End Sub
